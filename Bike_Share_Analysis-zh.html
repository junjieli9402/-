def time_of_trip(datum, city):
    """
    
    
    
    
    
    
     datetime 
    
     https://docs.python.org/3/library/datetime.html#strftime-and-strptime-behavior
    """
    
    # 
    if city == 'NYC':
        starttime = datetime.strptime(datum['starttime'], '%m/%d/%Y %X')

    elif city =='Chicago':
        starttime = datetime.strptime(datum['starttime'], '%m/%d/%Y %H:%M')
        
    elif city == 'Washington':
        starttime = datetime.strptime(datum['Start date'], '%m/%d/%Y %H:%M')
    
    else:
         starttime = datetime.strptime(datum['Start Date'], '%m/%d/%Y %H:%M')
        
    month = starttime.month
    hour = starttime.hour
    weekday = starttime.weekday() 
    
    day_of_week_list = ['Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday', 'Sunday']
    day_of_week = day_of_week_list[weekday]

        
    
    return (month, hour, day_of_week)

def condense_data(in_file, out_file, city):
    """
    
    
    
    
    
    """
    
    with open(out_file, 'w', newline = '') as f_out, open(in_file, 'r') as f_in:
        #  csv DictWriter 
        #  "fieldnames" 
        out_colnames = ['duration', 'month', 'hour', 'day_of_week', 'user_type']        
        trip_writer = csv.DictWriter(f_out, fieldnames = out_colnames)
        trip_writer.writeheader()
        
        ##  csv DictReader ##
        trip_reader = csv.DictReader(f_in)
    
        # 
        for row in trip_reader:
            # 
            new_point = {}
            
            ## ##

            ###'duration'
            
            new_point['duration'] = duration_in_mins(row, city)
            new_point['month'], new_point['hour'], new_point['day_of_week'] = time_of_trip(row, city)
            new_point['user_type'] = type_of_user(row, city)
            ##  new_point  ##
            ##  DictWriter         ##
            

            ## ##
            ##  https://docs.python.org/3/library/csv.html#writer-objects ##
            trip_writer.writerow(new_point)
        return trip_writer

def number_of_trips(filename):
    """
    
    
    """
    with open(filename, 'r') as f_in:
        #  csv reader 
        reader = csv.DictReader(f_in)
        
        # 
        n_subscribers = 0
        n_customers = 0
        
        # 
        for row in reader:
            if row['user_type'] == 'Subscriber':
                n_subscribers += 1
            else:
                n_customers += 1
        
        # 
        n_total = n_subscribers + n_customers
        # 
        sub_perc = n_subscribers/n_total
        # 
        cus_perc = n_customers/n_total
        
        # 
        return(n_total, sub_perc, cus_perc)

            trips_duration += float(row['duration'])

            if float(row['duration']) > 30:
                halfhour_number += 1
                
                # 30
                avg_dur = trips_duration/row_number
